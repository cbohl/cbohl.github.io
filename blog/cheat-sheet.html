Simultaneous Execution


var terah = {
  name: "Terah",
  age: 32,
  height: 66,
  weight: 130,
  hairColor: "brown",
  eyeColor: "brown"
}

var adam = {
  name: "Adam",
  spouse: terah,
}

terah.spouse = adam;
terah.weight = 125;
delete terah.eyeColor;




terah.children = {
  carson: {name: "Carson"},
  carter: {name: "Carter"},
  colton: {name: "Colton"},
}
adam.children = terah.children;
